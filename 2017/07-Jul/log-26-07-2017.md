### Array.from
_Array.from transforma todos os elementos similares ao array em Object Array, para que ele possa receber todos os mÃ©todos e propriedades do prototype do array_

```javascript
const text = "William"
console.log(Array.from(text))
// ["W", "i", "l", "l", "i", "a", "m"]

/* Markup example
<ul id="list">  
  <li>William</li>  
  <li>Barbara</li>  
  <li>Maria</li>  
  <li>Sergio</li>  
  <li>Well</li>  
</ul>  
*/

const list = document.querySelectorAll('#list li')
const listArrayFrom = Array.from(list)

console.log(list) // Without Array Properties
console.log(listArrayFrom) // With Array Properties

// .map array
const names = listArrayFrom.map((name) => name.textContent)
console.log(names); // ["William", "Barbara", "Maria", "Sergio", "Well"]
```
